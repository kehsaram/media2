// Firebase Storage Security Rules
// Copy these rules to your Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Rules for media files
    match /media/{userId}/{mediaType}/{fileName} {
      // Allow read/write access only to authenticated users for their own files
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Additional validation for uploads
      allow create: if request.auth != null 
                    && request.auth.uid == userId
                    && isValidMediaType(mediaType)
                    && isValidFileSize(request.resource.size)
                    && isValidContentType(request.resource.contentType);
      
      // Allow updates only to metadata, not file content
      allow update: if request.auth != null 
                    && request.auth.uid == userId
                    && request.resource.size == resource.size;
      
      // Allow delete
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Deny access to any other paths
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}

// Helper functions
function isValidMediaType(mediaType) {
  return mediaType in ['image', 'video', 'audio', 'document'];
}

function isValidFileSize(size) {
  // Maximum file size: 100MB (adjust as needed)
  return size < 100 * 1024 * 1024;
}

function isValidContentType(contentType) {
  return contentType.matches('image/.*') ||
         contentType.matches('video/.*') ||
         contentType.matches('audio/.*') ||
         contentType.matches('application/.*') ||
         contentType.matches('text/.*');
}
