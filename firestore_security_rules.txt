// Firestore Security Rules
// Copy these rules to your Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Rules for users collection
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Rules for media collection
    match /media/{documentId} {
      // Allow users to read/write only their own media documents
      allow read, write: if request.auth != null 
                         && request.auth.uid == resource.data.uploadedBy;
      
      // Allow create if user is authenticated and they're setting themselves as uploader
      allow create: if request.auth != null 
                    && request.auth.uid == request.resource.data.uploadedBy
                    && isValidMediaDocument(request.resource.data);
    }
    
    // Deny access to any other collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

// Helper functions
function isValidMediaDocument(data) {
  return data.keys().hasAll(['fileName', 'downloadUrl', 'mediaType', 'uploadedBy']) &&
         data.mediaType in ['image', 'video', 'audio', 'document'] &&
         data.uploadedBy is string &&
         data.fileName is string &&
         data.downloadUrl is string;
}
